# build_instructions.txt
# Minimal build instructions for RPI_PICO Micropython with module manifest.py.

# Clone Micropython repository.
  git clone https://github.com/micropython/micropython.git
  cd micropython

# Navigate to ports/rp2.
# Create manifest.py in the rp2 directory.
# Add all custom modules to modules directory and include all modules in that directory in manifest.

# minimal manifest.py
  module("example_module.py")
# See firmware/port/rp2/manifest.py for full manifest.
# End of manifest.

# Do not forget to include _boot, _boot_fat, and rp2 or you will not have filesystem access once flashed!
# Inclusion of asb, asb_fman, and mfrc522 are optional but any excluded modules must be present in /lib on board fs.
# For 'keys.json' integrity assurance asb_crypt needs to be frozen in firmware. It will work if included in /lib
# but the stored hash will be exposed and readable/writable, allowing an attacker to override the integrity check.

# Add modules, commit changes, create tag, make submodules, clean build directory, and build with frozen modules.
  git status
  git add -A
  git commit
  NEWTAG="v1.29-pre-ASB-vX.X.X"
  git tag -f "$NEWTAG" HEAD
  make BOARD=RPI_PICO submodules
  make BOARD=RPI_PICO MANIFEST=manifest.py clean
  make BOARD=RPI_PICO MANIFEST=manifest.py

# After building, firmware UF2 and/or BIN will be located at /ports/rp2/BUILD-RPI_PICO pr ports/yourport/BOARD-"NAME".
